  <select id="listByPhoneAndTime" resultMap="WithPhoneMap">
	<!-- 只选取需要的字段 -->
  	select ci_tab.cms_call_id, ci_tab.type, ci_tab.begin_time, ci_tab.stop_time, 
  		ci_tab.call_duration, cca1.phone_num from_phone, cca2.phone_num to_phone
  	from 
		<!-- 先查符合条件的通话记录，把查询结果作为一个临时表 -->
	  	(select cci.cms_call_id, cci.type, cci.begin_time,
	  		cci.stop_time, cci.call_duration, cci.from_id, cci.to_id
	  	from cms_call_info cci, cms_customer_cost ccc, cms_customer_auth cca
	  	where cca.phone_num = #{phone } and
	  		  ccc.cms_cc_rid = cca.cms_auth_id and
			  
			  <!-- 不管是作为接收方，还是作为拨打方，都查 -->
	  		  (cci.from_id = ccc.cms_cc_id or cci.to_id = ccc.cms_cc_id) and
			  
			  <!-- 需处于给定期间的通话记录 -->
	  		  cci.begin_time &gt; #{beginTime} and
	  		  cci.begin_time &lt; #{endTime}) as ci_tab
	  		  
		<!-- 把上述临时表连续与客户消费表和客户认证表作关联，关联字段为拨打方id，进而获得拨打方phone -->
	  	left outer join cms_customer_cost ccc1 on ci_tab.from_id = ccc1.cms_cc_id
	  	left outer join cms_customer_auth cca1 on ccc1.cms_cc_rid = cca1.cms_auth_id
	  	
	  	<!-- 同理，获得接收方phone -->
    	left outer join cms_customer_cost ccc2 on ci_tab.to_id = ccc2.cms_cc_id
    	left outer join cms_customer_auth cca2 on ccc2.cms_cc_rid = cca2.cms_auth_id
  </select>
